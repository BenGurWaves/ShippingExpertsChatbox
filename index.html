async function sendMessage() { const input = document.getElementById("userInput"); const message = input.value; input.value = ""; // Show user message const chat = document.getElementById("chat"); chat.innerHTML += `
You: ${message}
`; // Call backend const res = await fetch("/.netlify/functions/chat", { method: "POST", body: JSON.stringify({ message }) }); const data = await res.json(); // Show AI reply chat.innerHTML += `
Bot: ${data.reply}
`; }
